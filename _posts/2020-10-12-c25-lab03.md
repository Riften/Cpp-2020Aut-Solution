### 第1关：打印三角形
```cpp
#include <iostream>

using namespace std;

int main(int argc, char const *argv[])
{
    int height;
    char direction;
    cin >> height >> direction;
    if (direction == 'A') {
        for (int i = 0; i < height; i++) {
            /*** Your code here ***/
            for (int j = 0; j < height - 1 - i; j++) {
                cout << " ";
            }
            for (int j = 0; j < 2 * i + 1; j++) {
                cout << "*";
            }
            cout << endl;
            /******** End *********/
        }
    } else if (direction == 'V') {
        for (int i = 0; i < height; i++) {
            /*** Your code here ***/
            for (int j = 0; j < i; j++) {
                cout << " ";
            }
            for (int j = 0; j < 2 * height - 1 - 2 * i; j++) {
                cout << "*";
            }
            cout << endl;
            /******** End *********/
        }
    }
    return 0;
}
```

### 第3关：最大公约数
```cpp
#include <iostream>

using namespace std;

int main(int argc, char const *argv[])
{
    /*** Your code here ***/
    int m, n;
    cin >> m >> n;
    while (m != 0 && n != 0) {
        if (m > n) {
            m %= n;
        } else {
            n %= m;
        }
    }
    if (m == 0 && n == 0) {
        cout << "inf";
    } else if (m == 0) {
        cout << n;
    } else {
        cout << m;
    }
    /******** End *********/

}
```

### 第4关：方程求根
```cpp
#include <iostream>
#include <iomanip>
#include <cmath>

using namespace std;

int main(int argc, char const *argv[])
{
    cout.setf(ios::fixed);
    const long double delta = 1e-9;
    /*** Your code here ***/
    long double a, b, c, x0;
    cin >> a >> b >> c >> x0;
    long double f = x0 * x0 * x0 + a * x0 * x0 + b * x0 + c;
    long double df = 3.0 * x0 * x0 + 2.0 * a * x0 + b;
    while (fabs(f) >= delta) {
        x0 -= f / df;
        f = x0 * x0 * x0 + a * x0 * x0 + b * x0 + c;
        df = 3.0 * x0 * x0 + 2.0 * a * x0 + b;
    }
    cout << setprecision(3) << x0;
    /******** End *********/
    return 0;
}
```

### 第5关：算术平方根（附加题）
```cpp
#include <iostream>
#include <iomanip>
#include <cmath>

using namespace std;

int main(int argc, char const *argv[])
{
    /*** Your code here ***/
    cout.setf(ios::fixed);
    const long double delta = 1e-9;
    long double x, x0 = 1.0;
    cin >> x;
    long double f = x0 * x0 - x;
    long double df = 2.0 * x0;
    while (fabs(f) >= delta) {
        x0 -= f / df;
        f = x0 * x0 - x;
        df = 2.0 * x0;
    }
    cout << setprecision(3) << x0;
    /******** End *********/
    return 0;
}
```
