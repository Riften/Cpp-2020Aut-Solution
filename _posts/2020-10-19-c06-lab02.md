### 第4关：计算电费（纠错题）
```cpp
#include <iostream>

using namespace std;

int main() {
    // 变量定义：
    // - quantity: 所用电量（度）
    // - charge: 所需电费（元）
    // - yuan: 需要1元硬币数量
    // - jiao：需要1角硬币数量
    int quantity, yuan, jiao;
    int charge;
    
    cin >> quantity;              // 输入用电量
    charge = 6*quantity;          // 计算电费
    yuan = charge / 10;           // 计算需要几个1元硬币
    jiao = charge % 10;           // 计算需要几个1角硬币

    cout  << yuan << ' ' << jiao;   // 输出
    return 0;
}
```

### 第8关：方程求根
```cpp
#include <iostream>
#include <iomanip>
#include <cmath>

using namespace std;

int main(int argc, char const *argv[])
{
    cout.setf(ios::fixed);
    const long double delta = 1e-9;
    /*** Your code here ***/
    long double a, b, c, x0;
    cin >> a >> b >> c >> x0;
    long double f = x0 * x0 * x0 + a * x0 * x0 + b * x0 + c;
    long double df = 3.0 * x0 * x0 + 2.0 * a * x0 + b;
    while (fabs(f) >= delta) {
        x0 -= f / df;
        f = x0 * x0 * x0 + a * x0 * x0 + b * x0 + c;
        df = 3.0 * x0 * x0 + 2.0 * a * x0 + b;
    }
    cout << setprecision(3) << x0;
    /******** End *********/
    return 0;
}
```

### 第9关：算术平方根（附加题）
```cpp
#include <iostream>
#include <iomanip>
#include <cmath>

using namespace std;

int main(int argc, char const *argv[])
{
    /*** Your code here ***/
    cout.setf(ios::fixed);
    const long double delta = 1e-9;
    long double x, x0 = 1.0;
    cin >> x;
    long double f = x0 * x0 - x;
    long double df = 2.0 * x0;
    while (fabs(f) >= delta) {
        x0 -= f / df;
        f = x0 * x0 - x;
        df = 2.0 * x0;
    }
    cout << setprecision(3) << x0;
    /******** End *********/
    return 0;
}
```
