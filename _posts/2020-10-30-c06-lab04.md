## 第3关：回文字符串
```cpp
#include <iostream>

using namespace std;

int main(int argc, char const *argv[])
{
    /*** Your code here ***/
    int n;
    cin >> n;
    char str[1001];
    cin >> str[0];
    for (int i = 1; i < n; i++) {
        cin >> noskipws >> str[i];
    }
    int i = 0, j = n - 1;
    int DELTA = 'a' - 'A';
    while (i < j) {
        while ((str[i] < 'a' || str[i] > 'z') && (str[i] < 'A' || str[i] > 'Z')) {
            i++;
        }
        while ((str[j] < 'a' || str[j] > 'z') && (str[j] < 'A' || str[j] > 'Z')) {
            j--;
        }
        int delta = str[i] - str[j];
        if (delta == 0 || delta == DELTA || delta == -DELTA) {
            i++;
            j--;
        } else {
            break;
        }
    }
    if (i >= j) {
        cout << "yes";
    } else {
        cout << "no";
    }
    /******** End *********/
    return 0;
}
```

## 第4关：求解定积分

需要注意的是，在题干中给出的公式中，累加是从`i=1$`开始的，所以代码的循环中也应该从1开始循环。
```cpp
#include <iostream>
#include <iomanip>
#include <cmath>

using namespace std;

int main(int argc, char const *argv[])
{
    int n, k;
    double a, b;
    cin >> n >> a >> b >> k;
    int exps[100];        // 指数数组
    double coefs[100];    // 系数数组
    double delta = (b - a) / n;
    /*** Your code here ***/
    for (int i = 0; i < k; i++) {
        cin >> exps[i] >> coefs[i];
    }
    double sum = 0;
    for (int i = 1; i <= n; i++) {
        double x = a + i * delta;
        for (int j = 0; j < k; j++) {
            sum += coefs[j] * pow(x, exps[j]);
        }
    }
    sum *= delta;
    cout << setiosflags(ios::fixed) << setprecision(3) << sum;
    /******** End *********/
    return 0;
}
```

其实本题的评测脚本中，允许存在1%的误差，因此直接使用定积分公式也是可以通过评测的，感兴趣的同学可以尝试。

## 第5关：寻找鞍点
```cpp
#include <iostream>

using namespace std;

int main(int argc, char const *argv[])
{
    /*** Your code here ***/
    int m, n;
    cin >> m >> n;
    int mat[10][10];
    for (int i = 0; i < m; i++) {
      for (int j = 0; j < n; j++) {
        cin >> mat[i][j];
      }
    }

    bool exist_saddle = false;
    for (int i = 0; i < m; i++) {
      int col = 0;
      for (int j = 1; j < n; j++) {
        if (mat[i][j] > mat[i][col]) {
          col = j;
        }
      }

      bool saddle = true;
      for (int row = 0; row < m; row++) {
        if (mat[row][col] < mat[i][col]) {
          saddle = false;
          break;
        }
      }
      if (saddle) {
        exist_saddle = true;
        cout << i << " " << col << endl;
      }
    }

    if (!exist_saddle) {
      cout << "Not Found" << endl;
    }
    /******** End *********/
    return 0;
}
```
