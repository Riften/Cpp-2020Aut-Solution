### 第1关：三角形（改错题）
```cpp
#include <iostream>
#include <iomanip>
#include <cmath>

using namespace std;

int main()
{
    double a, b, c, s, area;

    cin >> a >> b >> c;

    if (a + b > c && a + c > b && b + c > a)
    {
        s = (a + b + c) / 2;
        area = sqrt(s * (s - a) * (s - b) * (s - c));
        cout << setprecision(2) << fixed << area << endl;
    }
    else
        cout << "Error" << endl;

    return 0;
}
```

### 第2关：计算 e（改错题）
```cpp
#include <iostream>
#include <iomanip>

using namespace std;

const double EPS = 1E-6; // 精度

int main()
{
    int i, n, item;
    double e;

    e = 1;
    n = 1;
    item = 1;

    do {
        item *= n;
        e += 1. / item;
        n++;
    } while (1. / item >= EPS);

    cout << fixed << setprecision(6) << e << endl;

    return 0;
}
```

### 第4关：换零钱
```cpp
#include <iostream>

using namespace std;

int main(int argc, char const *argv[])
{
    /*** Your code here ***/
    int n;
    cin >> n;
    n -= 8;
    int k = n / 5;
    int x = n % 5;
    int sum = 0;
    for (int i = 0; i <= k; i++) {
        sum += 1 + (x + 5 * i) / 2;
    }
    cout << sum;
    /******** End *********/
    return 0;
}
```
