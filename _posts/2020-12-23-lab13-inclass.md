## 第1关：温度计类

### thermometer.h
```cpp
#ifndef _THERMOMETER_H
#define _THERMOMETER_H

class Thermometer
{
private:
    double celsius; // 摄氏度
    double fahrenheit; // 华氏度

public:
    Thermometer(double celsius);
    void setCelsius(double celsius);
    void setFahrenheit(double fahrenheit);
    void showCelsius();
    void showFahrenheit();
};

#endif
```

### thermometer.cpp

```cpp
#include <iostream>
#include <iomanip>
#include "thermometer.h"

using namespace std;

// 构造函数
Thermometer::Thermometer(double celsius) {
    this->celsius = celsius;
    this->fahrenheit = 32.0 + celsius * 1.8;
}

// 使用如下方式实现成员函数
// void Thermometer::setCelsius(double celsius) {
//     ...
// }

/*** Your code here ***/
void Thermometer::setCelsius(double celsius) {
    this->celsius = celsius;
    this->fahrenheit = 32.0 + celsius * 1.8;
}

void Thermometer::setFahrenheit(double fahrenheit) {
    this->fahrenheit = fahrenheit;
    this->celsius = (fahrenheit - 32.0) / 1.8;
}

void Thermometer::showCelsius() {
    cout << fixed << setprecision(2) << this->celsius << endl;
}

void Thermometer::showFahrenheit() {
    cout << fixed << setprecision(2) << this->fahrenheit << endl;
}
```

### clock.cpp

```cpp
#include "clock.h"
#include <iostream>
using namespace std;
/*** Your code here ***/
void Clock::setTime(int hour, int minute, int second) {
    _hour = hour;
    _minute = minute;
    _second = second;
}

void Clock::increase(int second) {
    _second += second;
    _minute += _second / 60;
    _second %= 60;
    _hour += _minute / 60;
    _minute %= 60;
    _hour %= 24;
}

void Clock::print() {
    cout << _hour / 10 << _hour % 10 << ':' << _minute / 10 << _minute % 10 << ':' << _second / 10 << _second % 10 << endl;
}
```
