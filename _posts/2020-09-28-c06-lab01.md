### 第2关：读入两个整数，计算并输出它们的和、积、商和余数
```cpp
#include<iostream>

using namespace std;

int main() 
{
    // 请在此添加代码
    /********** Begin *********/
    int a, b;
    cin >> a >> b;
    cout << a << "+" << b << "=" << a + b << "\n";
    cout << a << "*" << b << "=" << a * b << "\n";
    cout << a << "/" << b << "=" << a / b << "\n";
    cout << a << "%" << b << "=" << a % b << "\n";
    /********** End **********/
  return 0;
}
```

### 第3关：读入一个三位整数num，求出num的百位数字、十位数字和个位数字
```cpp
#include<iostream>

using namespace std;

int main() 
{
    // 请在此添加代码
    /********** Begin *********/
    int num;
    cin >> num;
    cout << num << "的百位数字是" << num / 100
                << "，十位数字是" << (num % 100) / 10
                << "，个位数字是" << num % 10;
    /********** End **********/
    return 0;
}
```

### 第4关：输入一段时间的长度，计算该时间有多少小时多少分多少秒
```cpp
#include<iostream>

using namespace std;

int main() 
{

    // 请在此添加代码
    /********** Begin *********/
    int seconds;
    cin >> seconds;
    cout << seconds << "秒为";
    cout << seconds / 3600 << "小时";
    seconds %= 3600;
    cout << seconds / 60 << "分";
    seconds %= 60;
    cout << seconds << "秒\n";
    /********** End **********/
    return 0;
}
```
