### 第2关：反向输出字符串
```cpp
#include <iostream>

using namespace std;

int main(int argc, char const *argv[])
{
    /**
     * 实例代码给出了将字符串长度输入给n，并将字符串内容输入给str的方法。
     * 需要注意的是，对于字符变量x，直接使用 cin >> x; 则会跳过空格、制表符、换行符等空白字符。
     * 所以需要使用 cin >> noskipws >> x; 将空白字符输入给x。
     */
    int n;
    cin >> n;
    char str[1000];
    cin >> str[0];                  // 忽略字符串长度和字符串内容之间的空白字符（空格、制表符、换行符等）
    for (int i = 1; i < n; i++) {
        cin >> noskipws >> str[i];  // 实际字符串通过noskipws不忽略空白字符
    }
    /*** Your code here ***/
    for (int i = n - 1; i >= 0; i--) {
        cout << str[i];
    }
    /******** End *********/
    return 0;
}
```

### 第3关：回文字符串
```cpp
#include <iostream>

using namespace std;

int main(int argc, char const *argv[])
{
    /*** Your code here ***/
    int n;
    cin >> n;
    char str[1001];
    cin >> str[0];
    for (int i = 1; i < n; i++) {
        cin >> noskipws >> str[i];
    }
    int i = 0, j = n - 1;
    int DELTA = 'a' - 'A';
    while (i < j) {
        while ((str[i] < 'a' || str[i] > 'z') && (str[i] < 'A' || str[i] > 'Z')) {
            i++;
        }
        while ((str[j] < 'a' || str[j] > 'z') && (str[j] < 'A' || str[j] > 'Z')) {
            j--;
        }
        int delta = str[i] - str[j];
        if (delta == 0 || delta == DELTA || delta == -DELTA) {
            i++;
            j--;
        } else {
            break;
        }
    }
    if (i >= j) {
        cout << "yes";
    } else {
        cout << "no";
    }
    /******** End *********/
    return 0;
}
```

### 第4关：求解定积分
```cpp
#include <iostream>
#include <iomanip>
#include <cmath>

using namespace std;

int main(int argc, char const *argv[])
{
    int n, k;
    double a, b;
    cin >> n >> a >> b >> k;
    int exps[100];        // 指数数组
    double coefs[100];    // 系数数组
    double delta = (b - a) / n;
    /*** Your code here ***/
    for (int i = 0; i < k; i++) {
        cin >> exps[i] >> coefs[i];
    }
    double sum = 0;
    for (int i = 1; i <= n; i++) {
        double x = a + i * delta;
        for (int j = 0; j < k; j++) {
            sum += coefs[j] * pow(x, exps[j]);
        }
    }
    sum *= delta;
    cout << setiosflags(ios::fixed) << setprecision(3) << sum;
    /******** End *********/
    return 0;
}
```

        cin >> temp;
        a[i] = temp - '0';
    }
    cin >> n;
    int b[501];
    for (int i = n - 1; i >= 0; i--) {
        cin >> temp;
        b[i] = temp - '0';

    for (int i = 0; i < m; i++) {
        for (int j = 0; j < n; j++) {
            c[i + j] += a[i] * b[j];
        }
    }
    for (int i = 0; i < m + n - 1; i++) {
        c[i + 1] += c[i] / 10;
        c[i] %= 10;
    }
    int i = m + n - 1;
    for (; i >= 0; i--) {
        if (c[i]) {
            for (; i >=0 ; i--) {

        }
    }

}
